**Create /api/stripe/connect/webhook/route.ts:**

Stripe webhook endpoint for `account.updated` events. When a connected account completes onboarding (`details_submitted: true`), confirm in Convex.

```typescript
import Stripe from "stripe";
import { NextRequest, NextResponse } from "next/server";
import { ConvexHttpClient } from "convex/browser";
import { internal } from "@/convex/_generated/api";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
const convex = new ConvexHttpClient(process.env.NEXT_PUBLIC_CONVEX_URL!);

export async function POST(req: NextRequest) {
  const body = await req.text();
  const sig = req.headers.get("stripe-signature");

  if (!sig) {
    return NextResponse.json({ error: "Missing signature" }, { status: 400 });
  }

  let event: Stripe.Event;
  try {
    event = stripe.webhooks.constructEvent(
      body,
      sig,
      process.env.STRIPE_CONNECT_WEBHOOK_SECRET!
    );
  } catch (err) {
    console.error("[stripe/connect/webhook] Signature verification failed:", err);
    return NextResponse.json({ error: "Invalid signature" }, { status: 400 });
  }

  if (event.type === "account.updated") {
    const account = event.data.object as Stripe.Account;
    if (account.details_submitted) {
      convex.setAdminAuth(process.env.CONVEX_DEPLOY_KEY!);
      await convex.mutation(
        internal.launch.stripeConnect.confirmStripeConnect,
        { stripeConnectAccountId: account.id }
      );
    }
  }

  return NextResponse.json({ received: true });
}
```

Key details:
- Use `STRIPE_CONNECT_WEBHOOK_SECRET` (separate from any existing webhook secret — this is a new endpoint)
- Only process `account.updated` where `details_submitted === true`
- Use ConvexHttpClient with admin auth to call internal mutation
- Return 200 immediately for all events (Stripe expects this)
- Import `ConvexHttpClient` from `"convex/browser"` (the correct import path for ConvexHttpClient in Convex)
